.data
a: .word64 0
b: .word64 0
c: .word64 0
d: .word64 0

.code
; a = 2
daddiu r1, r0, 2
sd r1, a(r0)

; b = 4
daddiu r1, r0, 4
sd r1, b(r0)

; c = 8
daddiu r1, r0, 8
sd r1, c(r0)

; temp0 = a * c
ld r1, a(r0)
ld r2, c(r0)
dmult r1, r2
mflo r3

; temp1 = b + a
ld r1, b(r0)
ld r2, a(r0)
daddu r4, r1, r2

; temp2 = c / 2
ld r1, c(r0)
daddiu r2, r0, 2
ddiv r1, r2
mflo r5

; temp3 = temp1 - temp2
dsub r1, r4, r5

; d = temp0 / temp3
ddiv r3, r1
mflo r2
sd r2, d(r0)
